instance method Object#class

class -> Class[permalink][rdoc][edit]

レシーバのクラスを返します。 


p "ruby".class #=> String
p 100.class #=> Integer
p ARGV.class #=> Array
p self.class #=> Object
p Class.class #=> Class
p Kernel.class #=> Module
