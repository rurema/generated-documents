singleton method RakeFileUtils.verbose_flag

verbose_flag -> bool[permalink][rdoc][edit]

この値が真の場合、詳細を表示します。 


# Rakefile での記載例とする

task default: :sample_file_task

file :sample_file_task do |t|
  # --verbose で rake を実行する
  RakeFileUtils.verbose_flag # => true
end
