instance method OptionParser#summary_indent

summary_indent -> String[permalink][rdoc][edit]

サマリを表示する時のインデントを文字列で返します。

[RETURN]
サマリを表示する時のインデントを文字列で返します。 
例

require "optparse"

opts = OptionParser.new do |opts|
  opts.on_head("-i", "--init")
  opts.on("-u", "--update")
  opts.on_tail("-h", "--help")
end

opts.summary_indent         # => "    "
opts.summarize
# => ["    -i, --init\n", "    -u, --update\n", "    -h, --help\n"]
opts.summary_indent = "  "
opts.summary_indent         # => "  "
opts.summarize
# => ["  -i, --init\n", "  -u, --update\n", "  -h, --help\n"]
