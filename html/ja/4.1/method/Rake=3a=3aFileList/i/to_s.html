instance method Rake::FileList#to_s

to_s -> String[permalink][rdoc][edit]

全ての要素をスペースで連結した文字列を返します。 


# Rakefile での記載例とする

task default: :test_rake_app
task :test_rake_app do
  file_list = FileList['a.c', 'b.c']
  file_list.to_s # => "a.c b.c"
end
